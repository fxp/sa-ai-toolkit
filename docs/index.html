Get Started

# Clone the repo
$ git clone https://github.com/fxp/sa-ai-toolkit.git
# Install skills (copy to Claude Code)
$ cp -r demos/sa-ai-toolkit/skills/* ~/.claude/skills/
# Generate a demo package for any company
$ /gen "your target company"
# Try the Org-Uplift Game
$ cd demos/demo-metr-org-uplift && python3 -m http.server 8765
# Try the Industrial AI Pipeline
$ cd demos/demo-industrial-ai-pipeline && python3 pipeline.py run
